void ListaGenerica::insertar(int pos, Zapato x)
{
    if (pos <= cantidad() + 1)
    {
        Nodo* nuevo = new Nodo();
        nuevo->info = x;
        if (pos == 1)
        {
            nuevo->sig = raiz;
            raiz = nuevo;
        }
        else
            if (pos == cantidad() + 1)
            {
                Nodo* reco = raiz;
                while (reco->sig != NULL)
                {
                    reco = reco->sig;
                }
                reco->sig = nuevo;
                nuevo->sig = NULL;
            }
            else
            {
                Nodo* reco = raiz;
                for (int f = 1; f <= pos - 2; f++)
                {
                    reco = reco->sig;
                }
                Nodo* siguiente  = reco->sig;
                reco->sig = nuevo;
                nuevo->sig = siguiente;
            }
    }
}

                reco->sig = nuevo;

                nuevo->sig = siguiente;
